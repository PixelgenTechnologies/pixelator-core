42impl fmt::Debug for PartitionedGraphStatistics {
43    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
44        let mut sorted_partition_sizes = self.partition_node_counts.clone();
45        sorted_partition_sizes.sort_unstable_by(|a, b| b.partial_cmp(a).unwrap());
46        f.debug_struct("PartitionedGraphStatistics")
47            .field("current_edge_weight_sum", &self.current_edge_weight_sum)
48            .field("original_node_count", &self.original_node_count)
49            .field(
50                "fraction_nodes_in_largest_partition",
51                &self.fraction_nodes_in_largest_partition,
52            )
53            .field("stranded_nodes", &self.stranded_nodes)
54            .field("crossing_edge_weight_sum", &self.crossing_edge_weight_sum)
55            .field(
56                "partition_node_counts",
57                &format!(
58                    "[{:?},...; {}]",
59                    &sorted_partition_sizes[0..5],
60                    sorted_partition_sizes.len()
61                ),
62            )
63            .field("quality", &self.quality)
64            .finish()
65    }

91/// A graph with weighted nodes and a partitioning, used for Leiden community detection.
92///
93/// In particular, it provides the `aggregate` method that makes it possible to aggregate nodes
94/// belonging to the same partition as described in the Leiden algorithm.
95///
96/// As partitions are refined and nodes are aggregated, this structure also keeps track of the
97/// mapping from the original nodes (called ancestor nodes) to their current partitions,
98/// which will change as the graph is aggregated
99#[derive(Clone)]
100pub struct WeightedPartitionedGraph<Q: QualityMetrics> {
101    graph: Graph<usize>,
102    partitions: LeidenNodePartitioning,
103    node_weights: Vec<usize>,
104    partition_weights: HashMap<PartitionId, usize>,
105    quality_metrics: Q,
106    /// maps from ancestor node ids to current node ids
107    /// which will change as the graph is aggregated
108    ancestor_map: Vec<NodeIdx>,
109    rng: StdRng,
110}
111
112impl<Q: QualityMetrics> WeightedPartitionedGraph<Q> {
113    /// Create a new WeightedPartitionedGraph
114    ///
115    /// # Arguments
116    /// - `graph`: The underlying graph structure.
117    /// - `partitions`: The current partitioning of the nodes.
118    /// - `quality_metrics`: The quality metrics used for community detection. Defines how to
119    ///   compute node sizes and resolution.
120    /// - `node_weights`: Optional weights for each node. If None, weights are computed using using
121    ///   `quality_metrics.node_size`.
122    /// - `seed`: Optional seed for random number generation. If None, a default seed of 0 is used.
123    pub fn new(
124        graph: Graph<usize>,
125        partitions: LeidenNodePartitioning,
126        quality_metrics: Q,
127        node_weights: Option<Vec<usize>>,
128        seed: Option<u64>,
129    ) -> Self {
130        let node_weights = node_weights.unwrap_or_else(|| {
131            (0..graph.get_num_nodes())
132                .map(|node_id| quality_metrics.node_size(&graph, node_id))
133                .collect::<Vec<usize>>()
134        });
135
136        let mut partition_weights = HashMap::<PartitionId, usize>::default();
137        for (node_id, &partition_id) in partitions.get_node_to_partition_map().iter().enumerate() {
138            partition_weights
139                .entry(partition_id)
140                .and_modify(|e| *e += node_weights[node_id])
141                .or_insert(node_weights[node_id]);
142        }
143
144        let ancestor_map = (0..graph.get_num_nodes()).collect::<Vec<NodeIdx>>();
145        let rng = StdRng::seed_from_u64(seed.unwrap_or(0));
146
147        WeightedPartitionedGraph {
148            graph,
149            partitions,
150            node_weights,
151            partition_weights,
152            quality_metrics,
153            ancestor_map,
154            rng,
155        }
156    }

170    pub fn get_statistics(&self) -> PartitionedGraphStatistics {
171        let current_edge_weight_sum = self.graph.get_total_edge_weight();
172        let original_node_count = self.ancestor_map.len();
173
174        let partition_node_counts: Vec<usize> = self
175            .ancestor_map
176            .iter()
177            .map(|&node_id| self.partitions.get_node_to_partition_map()[node_id])
178            .counts()
179            .values()
180            .copied()
181            .collect();
182
183        let fraction_nodes_in_largest_partition =
184            partition_node_counts.iter().max().copied().unwrap_or(0) as f64
185                / original_node_count as f64;
186        let stranded_nodes = partition_node_counts
187            .iter()
188            .filter(|&&n_nodes| n_nodes < MIN_PNA_COMPONENT_SIZE)
189            .sum();
190
191        let crossing_edge_weight_sum = self
192            .graph
193            .get_edges_iter()
194            .filter(|Edge { src, dest, .. }| {
195                src <= dest
196                    && self.partitions.get_node_to_partition_map()[*src]
197                        != self.partitions.get_node_to_partition_map()[*dest]
198            })
199            .map(
200                |Edge {
201                     src: _,
202                     dest: _,
203                     weight,
204                 }| weight,
205            )
206            .sum();
207
208        PartitionedGraphStatistics {
209            current_edge_weight_sum,
210            original_node_count,
211            fraction_nodes_in_largest_partition,
212            stranded_nodes,
213            crossing_edge_weight_sum,
214            partition_node_counts,
215            quality: self.quality(),
216        }
217    }

219    /// Update the partition of a given node
220    ///
221    /// # Arguments
222    /// - `node`: The node index whose partition is to be updated.
223    /// - `new_partition`: The new partition id to assign to the node.
224    pub fn update_partition_slow(&mut self, node: NodeIdx, new_partition: PartitionId) {
225        let current_partition = self.partitions.get_partition_for_node(&node);
226        self.partition_weights
227            .entry(current_partition)
228            .and_modify(|e| *e -= self.node_weights[node]);
229        self.partition_weights
230            .entry(new_partition)
231            .and_modify(|e| *e += self.node_weights[node])
232            .or_insert(self.node_weights[node]);
233        self.partitions.update_partition(node, new_partition);
234    }

274    /// Normalize partitions so that all ids from 0 to |P| - 1 are used.
275    pub fn normalize_partition(&mut self) {
276        let partition_index = self
277            .partitions
278            .get_partition_to_node_map_ref()
279            .iter()
280            .filter(|(_, nodes)| !nodes.is_empty())
281            .enumerate()
282            .map(|(new_pid, (&old_pid, _))| (old_pid, new_pid))
283            .collect::<HashMap<PartitionId, PartitionId>>();
284
285        self.partitions = LeidenNodePartitioning::initialize_from_partitions(
286            self.partitions
287                .get_node_to_partition_map()
288                .iter()
289                .map(|pid| partition_index[pid])
290                .collect(),
291        );
292
293        self.partition_weights = self
294            .partition_weights
295            .iter()
296            .filter_map(|(pid, &weight)| {
297                if let Some(&new_pid) = partition_index.get(pid) {
298                    Some((new_pid, weight))
299                } else {
300                    None
301                }
302            })
303            .collect();
304    }

306    /// Aggregate nodes sharing the same partition
307    ///
308    /// If `aggregate_opts` is `AggregateOptions::OnlyWellConnected`, partitions will be refined
309    /// so that only well connected nodes are aggregated
310    pub fn aggregate(&mut self, aggregate_opts: AggregateOptions) {
311        let (super_partitions, super_partition_weights) = match aggregate_opts {
312            AggregateOptions::OnlyWellConnected(randomness) => {
313                let (super_partitions, super_partition_weights) =
314                    self.refine_partitions(randomness);
315                (Some(super_partitions), Some(super_partition_weights))
316            }
317            AggregateOptions::All => (None, None),
318        };
319
320        // filter out empty partitions, and normalize the partition ids from 0..n
321        // since this is necessary for building the aggregated graph
322        let mut normalized_partition_map = vec![0; self.partitions.get_new_partition_id()];
323        for (new_pid, (&old_pid, _)) in self
324            .partitions
325            .get_partition_to_node_map_ref()
326            .iter()
327            .filter(|(_, nodes)| !nodes.is_empty())
328            .enumerate()
329        {
330            normalized_partition_map[old_pid] = new_pid;
331        }
332        // Map node ids between the current graph and the aggregated graph
333        let sub_partition_index = self
334            .partitions
335            .get_node_to_partition_map()
336            .iter()
337            .map(|&p_id| normalized_partition_map[p_id])
338            .collect::<Vec<NodeIdx>>();
339
340        let aggregated_graph = self.build_aggregated_graph(&sub_partition_index);
341
342        let n_partitions = self.partitions.num_partitions();
343        let (new_partitions, new_node_weights) =
344            self.build_new_partitions(&sub_partition_index, super_partitions, n_partitions);
345
346        self.ancestor_map
347            .iter_mut()
348            .for_each(|node_id| *node_id = sub_partition_index[*node_id]);
349
350        self.graph = aggregated_graph;
351        self.partitions = new_partitions;
352        self.node_weights = new_node_weights;
353        if let Some(super_partition_weights) = super_partition_weights {
354            self.partition_weights = super_partition_weights;
355        }
356    }

358    /// Refine current partitions to separate poorly connected nodes and clusters into different
359    /// communities.
360    ///
361    /// The unrefined partitioning becomes super partitioning and is returned by this function,
362    /// together with the weights of the super partitions.
363    fn refine_partitions(
364        &mut self,
365        randomness: f64,
366    ) -> (LeidenNodePartitioning, HashMap<PartitionId, usize>) {
367        // Reset current partitions and weights to singlet partitions
368        let mut super_partitions =
369            LeidenNodePartitioning::initialize_with_singlet_partitions(self.graph.get_num_nodes());
370        mem::swap(&mut self.partitions, &mut super_partitions);
371
372        let mut super_partition_weights = self
373            .node_weights
374            .iter()
375            .enumerate()
376            .map(|(node_id, &weight)| (node_id, weight))
377            .collect::<HashMap<PartitionId, usize>>();
378        mem::swap(&mut self.partition_weights, &mut super_partition_weights);
379
380        // Group well connected nodes belonging to the same super partition
381        for (id, nodes) in super_partitions.get_partition_to_node_map_ref().iter() {
382            self.merge_node_subset(nodes, super_partition_weights[id], randomness);
383        }
384
385        (super_partitions, super_partition_weights)
386    }

388    /// Build new graph, aggregating nodes belonging to the same partitions into nodes
389    /// in a new graph.
390    ///
391    /// `partition_index` maps each node id in the original graph to the id of its aggregated
392    /// node (partition) in the new graph. The values are expected to be in the range
393    /// `0..num_aggregated_nodes` so they can be used directly as node indices.
394    fn build_aggregated_graph(&self, partition_index: &[NodeIdx]) -> Graph<usize> {
395        let num_aggregated_nodes = self.partitions.num_partitions();
396
397        Graph::from_edges(
398            self.graph
399                .get_edges_iter()
400                .filter(|Edge { src, dest, .. }| src <= dest)
401                .map(|Edge { src, dest, weight }| Edge {
402                    src: partition_index[src],
403                    dest: partition_index[dest],
404                    weight,
405                }),
406            num_aggregated_nodes,
407        )
408    }

410    /// Generate new partition table and partition weights, based on the new node ids.
411    ///
412    /// If the partitions have been refined, `super_partitions` can be used to specify the current
413    /// partitioning. Otherwise, singlet partitions will be assumed.
414    fn build_new_partitions(
415        &self,
416        sub_partition_index: &[PartitionId],
417        super_partitions: Option<LeidenNodePartitioning>,
418        n_partitions: usize,
419    ) -> (LeidenNodePartitioning, Vec<usize>) {
420        let mut new_partitions = vec![0; n_partitions];
421        let mut new_node_weights = vec![0; n_partitions];
422        izip!(
423            sub_partition_index.iter(),
424            self.partitions.get_node_to_partition_map().iter(),
425            super_partitions
426                .as_ref()
427                .unwrap_or(&self.partitions)
428                .get_node_to_partition_map()
429                .iter(),
430        )
431        .for_each(|(&new_pid, refined_pid, &super_pid)| {
432            new_partitions[new_pid] = super_pid;
433            new_node_weights[new_pid] = self.partition_weights[refined_pid];
434        });
435
436        (
437            LeidenNodePartitioning::initialize_from_partitions(new_partitions),
438            new_node_weights,
439        )
440    }

442    /// Sort by partition and aggregate edges outgoing from `node`
443    ///
444    /// This function is typically called millions of time in a loop. To avoid repeatedly
445    /// allocating a new vector each time, the return vector `aggregated_edge_weights` should be
446    /// allocated outside the function and passed as a mutable reference.
447    ///
448    /// Given that the nodes we deal with typically have only a few edges (median is 3, 99th
449    /// percentile is 22). The overhead of using a HashMap is prohibitive in this case.
450    ///
451    /// # Arguments
452    /// - `node`: node whose edges are being aggregated
453    /// - `aggregate_edge_weights`: variables where the aggregated edges will be stored.
454    fn aggregate_edge_weights(
455        &self,
456        node: NodeIdx,
457        aggregated_edge_weights: &mut Vec<(PartitionId, usize)>,
458    ) {
459        aggregated_edge_weights.clear();
460
461        for Edge { src, dest, weight } in self.graph.edges_from_iter(node) {
462            if src != dest {
463                let dst_partition = self.partitions.get_partition_for_node(&dest);
464
465                // Find the partition or push a new one.
466                // For small degrees (e.g. < 50), this destroys a HashMap.
467                match aggregated_edge_weights
468                    .iter_mut()
469                    .find(|(p, _)| *p == dst_partition)
470                {
471                    Some((_, total_weight)) => *total_weight += weight,
472                    None => aggregated_edge_weights.push((dst_partition, weight)),
473                }
474            }
475        }
476    }

478    /// Merge well connected nodes from the given subset
479    ///
480    /// This follows the implementation from the original Leiden description.
481    fn merge_node_subset(
482        &mut self,
483        node_subset: &HashSet<NodeIdx>,
484        node_subset_size: usize,
485        randomness: f64,
486    ) {
487        // First consider only nodes that are well connected within the subset
488        let mut well_connected_nodes = node_subset
489            .iter()
490            .filter(|&&node_id| self.node_is_well_connected(node_id, node_subset, node_subset_size))
491            .cloned()
492            .collect::<Vec<NodeIdx>>();
493
494        // Visit them in random order
495        well_connected_nodes.shuffle(&mut self.rng);
496
497        let mut aggregated_edge_weights: Vec<(PartitionId, usize)> = Vec::new();
498        let mut community_deltas: Vec<(PartitionId, f64)> = Vec::new();
499        for node in well_connected_nodes.iter() {
500            // Only consider moving nodes that are currently in singlet partitions
501            // as we are calling this from the  refinement step, all nodes will
502            // start in singlet partitions, but as we update this they will potentially
503            // end up in larger partitions.
504            if self
505                .partitions
506                .is_singlet_partition(&self.partitions.get_partition_for_node(node))
507            {
508                community_deltas.clear();
509                let node_size = self.node_weights[*node];
510
511                // Find the communities of the neighbors of node, since we can only transfer
512                // to those communities
513                self.aggregate_edge_weights(*node, &mut aggregated_edge_weights);
514
515                let src_partition = self.partitions.get_partition_for_node(node);
516                let src_size = self.partition_weights[&src_partition];
517                let src_weight = aggregated_edge_weights
518                    .iter()
519                    .find(|(p, _)| *p == src_partition)
520                    .map(|(_, w)| *w)
521                    .unwrap_or(0);
522                let loss_src = self.loss_src(node_size, src_weight, src_size);
523
524                community_deltas.extend(
525                    aggregated_edge_weights
526                        .iter()
527                        .filter(|(community_id, _)| {
528                            self.community_is_well_connected(
529                                community_id,
530                                node_subset,
531                                node_subset_size,
532                            )
533                        })
534                        .map(|&(community_id, total_edge_weight)| {
535                            (
536                                community_id,
537                                loss_src
538                                    + self.contrib_dest(
539                                        node_size,
540                                        total_edge_weight,
541                                        self.partition_weights[&community_id],
542                                    ),
543                            )
544                        })
545                        .filter(|&(_, delta)| delta > 0.),
546                );
547
548                // If none of the deltas are positive, we will not move anything
549                if community_deltas.is_empty() {
550                    continue;
551                }
552                // And then pick a random one from the weighted distribution of those that
553                // have a positive modularity delta
554                let partition_idx = pick_random_community_from_weighted_options(
555                    &community_deltas,
556                    randomness,
557                    &mut self.rng,
558                );
559                self.update_partition_slow(*node, partition_idx);
560            }
561        }
562    }

564    /// Select the community that gives the best quality increase for the given node.
565    ///
566    /// This function is typically called millions of time in a loop. To avoid repeatedly
567    /// allocating a new vector each time, the return vector `aggregated_edge_weights` should be
568    /// allocated outside the function and passed as a mutable reference.
569    pub fn best_community_for_node(
570        &self,
571        node: NodeIdx,
572        aggregated_edge_weights: &mut Vec<(PartitionId, usize)>,
573    ) -> Option<(PartitionId, f64)> {
574        self.aggregate_edge_weights(node, aggregated_edge_weights);
575
576        // If node has no neighbors
577        if aggregated_edge_weights.is_empty() {
578            return None;
579        }
580
581        let node_size = self.node_weights[node];
582        let src_partition = self.partitions.get_partition_for_node(&node);
583
584        aggregated_edge_weights
585            .iter()
586            .filter(|&&(partition, _)| partition != src_partition)
587            .filter_map(|&(partition, total_edge_weight)| {
588                let contrib = self.contrib_dest(
589                    node_size,
590                    total_edge_weight,
591                    self.partition_weights[&partition],
592                );
593                if contrib > 0.0 {
594                    Some((partition, contrib))
595                } else {
596                    None
597                }
598            })
599            .max_by(
600                |(community_id_a, contrib_dest_a), (community_id_b, contrib_dest_b)| {
601                    if is_close!(*contrib_dest_a, *contrib_dest_b) {
602                        community_id_a.cmp(community_id_b)
603                    } else {
604                        contrib_dest_a
605                            .partial_cmp(contrib_dest_b)
606                            .unwrap_or(std::cmp::Ordering::Equal)
607                    }
608                },
609            )
610            // A new community will always have 0 contribution since no neighbor will be a member
611            // of this community, and its size will be 0.
612            .or(Some((self.partitions.get_new_partition_id(), 0.0)))
613            .map(|(id, contrib_dest)| {
614                let src_size = self.partition_weights[&src_partition];
615                let src_weight = aggregated_edge_weights
616                    .iter()
617                    .find(|(p, _)| *p == src_partition)
618                    .map(|(_, w)| *w)
619                    .unwrap_or(0);
620                let loss_src = self.loss_src(node_size, src_weight, src_size);
621                (id, contrib_dest + loss_src)
622            })
623    }

625    /// Given a graph and a partitioning, computes the quality of this partitioning, based on the
626    /// implementation of the resolution and node size functions.
627    pub fn quality(&self) -> f64 {
628        self.partitions
629            .get_partition_to_node_map_ref()
630            .iter()
631            .map(|(&partition_id, nodes_in_partition)| {
632                let subset_size = self.partition_weights[&partition_id];
633                let edge_count_in_partition = self
634                    .graph
635                    .get_edges_in_selection(nodes_in_partition)
636                    .filter(|Edge { src, dest, .. }| {
637                        nodes_in_partition.contains(src) && nodes_in_partition.contains(dest)
638                    })
639                    .map(
640                        |Edge {
641                             src: _,
642                             dest: _,
643                             weight,
644                         }| weight,
645                    )
646                    .sum::<usize>();
647
648                let possible_pairs_in_community =
649                    (subset_size as f64 * (subset_size as f64 - 1.0)) / 2.0;
650                edge_count_in_partition as f64
651                    - self.quality_metrics.resolution() * possible_pairs_in_community
652            })
653            .sum()
654    }

656    #[allow(clippy::doc_overindented_list_items)]
657    #[allow(dead_code)]
658    /// Compute the quality delta induced by a given atomic change, i.e. one node moving from
659    /// one partition to another.
660    ///
661    /// The formula for quality is:
662    /// H(G, P) = ΣE(C,C) - γ|C|(|C| - 1)/2
663    ///
664    /// To compute the delta caused by moving a node from `src` to `dest`:
665    /// 1. Remove edges from target v to src, add edges from v to dest
666    ///    src  \|/   dest
667    ///         -v
668    ///         /|\
669    ///
670    /// 2. Add the changes from community sizes. The formula reads as follow: add back the
671    ///    contributions for both source and destination communities and then remove the
672    ///    contributions of the communities as they would be after target node v is moved from src
673    ///    to dest
674    ///
675    ///    In raw form this is:
676    ///    γ*(
677    ///           |src|*(|src| - 1)/2
678    ///         + |dst|*(|dst| - 1)/2
679    ///         - (|src| - |v|)*(|src| - |v| - 1)/2
680    ///         - (|dst| + |v|)*(|dst| + |v| - 1)/2
681    ///       )
682    ///
683    ///    Surprisingly this simplifies into
684    ///         γ * |v| * (|src| - |v| - |dst|)
685    ///
686    ///  When several destinations needs to be compared, steps 1. and 2. can be split further
687    ///  into the loss from src and the contribution from dest, see `loss_src` and `contrib_dest`
688    ///  below.
689    ///
690    ///  NB: calling this function repeatedly for all possible destination communities is quite
691    ///  inefficient, use `contrib_dest` and `loss_src` instead.
692    pub fn delta(&self, node_id: NodeIdx, dest_partition: PartitionId) -> f64 {
693        let node_size = self.node_weights[node_id];
694
695        let node_to_partition = self.partitions.get_node_to_partition_map();
696        let dest_edge_total_weight: usize = self
697            .graph
698            .edges_from_iter(node_id)
699            .filter(|Edge { src: _, dest, .. }| node_to_partition[*dest] == dest_partition)
700            .map(
701                |Edge {
702                     src: _,
703                     dest: _,
704                     weight,
705                 }| weight,
706            )
707            .sum();
708        let dest_size = self.get_partition_weights()[&dest_partition];
709
710        let src_partition = self.partitions.get_node_to_partition_map()[node_id];
711        let src_edge_total_weight: usize = self
712            .graph
713            .edges_from_iter(node_id)
714            .filter(|Edge { src: _, dest, .. }| node_to_partition[*dest] == src_partition)
715            .map(
716                |Edge {
717                     src: _,
718                     dest: _,
719                     weight,
720                 }| weight,
721            )
722            .sum();
723        let src_size = self.get_partition_weights()[&src_partition];
724        self.contrib_dest(node_size, dest_edge_total_weight, dest_size)
725            + self.loss_src(node_size, src_edge_total_weight, src_size)
726    }
727
728    fn loss_src(&self, node_size: usize, src_edge_total_weight: usize, src_size: usize) -> f64 {
729        -(src_edge_total_weight as f64)
730            + self.quality_metrics.resolution()
731                * (node_size as f64 * (src_size as f64 - node_size as f64))
732    }
733
734    fn contrib_dest(
735        &self,
736        node_size: usize,
737        dest_edge_total_weight: usize,
738        dest_size: usize,
739    ) -> f64 {
740        dest_edge_total_weight as f64
741            - self.quality_metrics.resolution() * node_size as f64 * dest_size as f64
742    }

744    /// Returns well connected nodes among the given subset.
745    ///
746    /// Well connected nodes satisfy the following formula (from A.2, l.34 in supplementary
747    /// information of the Leiden article <https://doi.org/10.1038/s41598-019-41695-z>):
748    ///
749    /// v ∈ S && E(v, S − v) ≥ γ ‖v‖ ⋅ (‖S‖ − ‖v‖)
750    pub fn node_is_well_connected(
751        &self,
752        node_id: NodeIdx,
753        node_subset: &HashSet<NodeIdx>,
754        subset_size: usize,
755    ) -> bool {
756        let node_size = self.node_weights[node_id];
757
758        let expected_connections_in_subset = self.quality_metrics.resolution()
759            * (node_size as f64 * (subset_size as f64 - node_size as f64));
760        // Count edges from `node_id` to nodes in `node_subset` excluding `node_id` itself.
761        let actual_edges_in_subset: f64 = self
762            .graph
763            .edges_from_iter(node_id)
764            .filter(|Edge { src: _, dest, .. }| *dest != node_id && node_subset.contains(dest))
765            .map(
766                |Edge {
767                     src: _,
768                     dest: _,
769                     weight,
770                 }| weight,
771            )
772            .sum::<usize>() as f64;
773
774        actual_edges_in_subset >= expected_connections_in_subset
775    }
776
777    /// Returns communities that are included and well connected to the supercommunity S.
778    ///
779    /// Such communities satisfy the following formula (from A.2, l.37 in supplementary information
780    /// of the Leiden article <https://doi.org/10.1038/s41598-019-41695-z>):
781    ///
782    /// C ∈ P && C ⊆ S && E(C, S − C) ≥ γ ‖C‖ ⋅ (‖S‖ − ‖C‖)
783    pub fn community_is_well_connected(
784        &self,
785        community_id: &PartitionId,
786        supercommunity: &HashSet<NodeIdx>,
787        supercommunity_size: usize,
788    ) -> bool {
789        let community_size = self.partition_weights[community_id];
790        let nodes_in_community = &self.partitions.get_partition_to_node_map_ref()[community_id];
791
792        // Check that all nodes in the community are inside the supercommunity
793        // If they are not, do not consider this community
794        if !nodes_in_community
795            .iter()
796            .all(|n| supercommunity.contains(n))
797        {
798            return false;
799        }
800
801        let actual_edge_count_in_community = self
802            .graph
803            .get_edges_in_selection(nodes_in_community)
804            .filter(|Edge { src, dest, .. }| {
805                (supercommunity.contains(src) && supercommunity.contains(dest))
806                    && (!nodes_in_community.contains(src) || !nodes_in_community.contains(dest))
807            })
808            .map(
809                |Edge {
810                     src: _,
811                     dest: _,
812                     weight,
813                 }| weight,
814            )
815            .sum::<usize>() as f64;
816        let expected_edges = (self.quality_metrics.resolution() * community_size as f64)
817            * (supercommunity_size as f64 - community_size as f64);
818
819        actual_edge_count_in_community >= expected_edges
820    }
821}

823fn pick_random_community_from_weighted_options<R: Rng>(
824    community_options: &[(PartitionId, f64)],
825    randomness: f64,
826    rng: &mut R,
827) -> PartitionId {
828    let total_weights = community_options.iter().map(|(_, w)| *w).sum::<f64>();
829    let weights = community_options
830        .iter()
831        .map(|(_, w)| (w / total_weights).powf(randomness))
832        .collect::<Vec<f64>>();
833    let dist = WeightedIndex::new(&weights).unwrap();
834
835    community_options[dist.sample(rng)].0
836}
