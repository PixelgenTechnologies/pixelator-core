1//! Graph statistics utilities.
2//!
3//! This module provides tools for computing basic statistics and properties of graphs,
4//! such as node and edge counts, number of connected components, and the fraction of nodes
5//! in the largest connected component.
6//!
7
8use crate::common::constants::MIN_PNA_COMPONENT_SIZE;
9use crate::common::{graph::Graph, types::EdgeWeight};
10use itertools::Itertools;
11use rustc_hash::FxHashMap as HashMap;
12
13/// Stores basic properties/statistics of a graph.
14#[derive(Debug, Clone)]
15pub struct GraphProperties {
16    /// Number of nodes in the graph.
17    pub node_count: usize,
18    /// Sum of undirected edge weights in the graph.
19    pub edge_weight_sum: usize,
20    /// Number of connected components in the graph.
21    pub n_connected_components: usize,
22    /// Fraction of nodes in the largest connected component.
23    pub fraction_in_largest_component: f64,
24    /// Number of nodes in components smaller than 8000 umis
25    pub stranded_nodes: usize,
26    /// Distribution of the component sizes
27    pub component_size_distribution: HashMap<usize, usize>,
28}
29
30impl GraphProperties {
31    /// Computes graph properties for the given graph.
32    pub fn new<T: EdgeWeight>(graph: &Graph<T>) -> Self {
33        let node_count = graph.get_num_nodes();
34        let edge_weight_sum = graph.get_total_edge_weight();
35
36        let component_sizes = graph
37            .connected_components()
38            .map(|nodes| nodes.len())
39            .collect::<Vec<_>>();
40        let component_count = component_sizes.len();
41
42        let fraction_in_largest_component =
43            component_sizes.iter().max().copied().unwrap_or(0) as f64 / node_count as f64;
44        let stranded_nodes = component_sizes
45            .iter()
46            .filter(|&&n_nodes| n_nodes < MIN_PNA_COMPONENT_SIZE)
47            .sum();
48        let component_size_distribution = component_sizes
49            .iter()
50            .copied()
51            .counts()
52            .into_iter()
53            .collect();
54
55        GraphProperties {
56            node_count,
57            edge_weight_sum,
58            n_connected_components: component_count,
59            fraction_in_largest_component,
60            stranded_nodes,
61            component_size_distribution,
62        }
63    }
64}

66#[cfg(test)]
67mod tests {
68
69    use super::*;
70    use crate::common::graph::Graph;
71    use crate::common::types::Edge;
72    use crate::common::types::edges_from_tuples;
73
74    #[test]
75    fn test_graph_properties_single_component() {
76        // Graph: 0-1-2
77        let edges = edges_from_tuples(vec![(0, 1), (1, 2)]);
78        let graph = Graph::<u8>::from_edges(edges.into_iter(), 3);
79        let props = GraphProperties::new(&graph);
80        assert_eq!(props.node_count, 3);
81        assert_eq!(props.edge_weight_sum, 2);
82        assert_eq!(props.n_connected_components, 1);
83        assert!((props.fraction_in_largest_component - 1.0).abs() < 1e-8);
84    }
85
86    #[test]
87    fn test_graph_properties_multiple_components() {
88        // Graph: 0-1  2 (isolated)
89        let edges = edges_from_tuples(vec![(0, 1)]);
90        let graph = Graph::<u8>::from_edges(edges.into_iter(), 3);
91        let props = GraphProperties::new(&graph);
92        assert_eq!(props.node_count, 3);
93        assert_eq!(props.edge_weight_sum, 1);
94        assert_eq!(props.n_connected_components, 2);
95        // Largest component: 0-1 (size 2), fraction = 2/3
96        assert!((props.fraction_in_largest_component - (2.0 / 3.0)).abs() < 1e-8);
97    }
98
99    #[test]
100    fn test_graph_properties_all_isolated() {
101        // Graph: 0 1 2 (all isolated)
102        let edges: Vec<Edge<u8>> = vec![];
103        let graph = Graph::<u8>::from_edges(edges.into_iter(), 3);
104        let props = GraphProperties::new(&graph);
105        assert_eq!(props.node_count, 3);
106        assert_eq!(props.edge_weight_sum, 0);
107        assert_eq!(props.n_connected_components, 3);
108        assert!((props.fraction_in_largest_component - (1.0 / 3.0)).abs() < 1e-8);
109    }
110}
