1use smallvec::SmallVec;
2
3use crate::common::graph::Graph;
4use crate::common::node_partitioning::{FastNodePartitioning, NodePartitioning};
5use crate::common::types::NodeIdx;
6use crate::common::utils::{NodeQueue, mode_via_sort};
7
8pub trait AssignmentStrategy<P> {
9    fn assign(
10        &self,
11        graph: &Graph<u8>,
12        node_partition: &mut P,
13        node_queue: &mut NodeQueue,
14        current_node: NodeIdx,
15    );
16}
17
18pub struct DefaultAssignmentStrategy;
19
20impl AssignmentStrategy<FastNodePartitioning> for DefaultAssignmentStrategy {
21    fn assign(
22        &self,
23        graph: &Graph<u8>,
24        node_partition: &mut FastNodePartitioning,
25        node_queue: &mut NodeQueue,
26        current_node: NodeIdx,
27    ) {
28        // Not using the built in neighbors method to avoid allocation of a Vec
29        let neighbors_view = graph
30            .get_adjacency_matrix()
31            .outer_view(current_node)
32            .unwrap();
33        if neighbors_view.nnz() == 0 {
34            return;
35        }
36        let neighbors = neighbors_view.indices();
37
38        // Using small vec to avoid heap allocation for small number of neighbors
39        // which should be the case for most of the nodes in the graph
40        let mut partitions = SmallVec::<[usize; 8]>::new();
41        partitions.extend(
42            neighbors
43                .iter()
44                .map(|x| node_partition.get_partition_for_node(x)),
45        );
46
47        // Finding the most common partition among neighbors by sorting and picking the mode
48        // from the vector is more efficient than using a hash map to count occurrences
49        // for thse small vectors we expect here.
50        let most_common_neighbors_partition = mode_via_sort(partitions.as_mut_slice());
51        let node_i_current_partition = node_partition.get_partition_for_node(&current_node);
52
53        if most_common_neighbors_partition != node_i_current_partition {
54            node_partition.update_partition(current_node, most_common_neighbors_partition);
55            for neighbor_to_update in neighbors
56                .iter()
57                .filter(|neighbor| {
58                    node_partition.get_partition_for_node(neighbor)
59                        != most_common_neighbors_partition
60                })
61                .copied()
62            {
63                node_queue.push_back(neighbor_to_update);
64            }
65        }
66    }
67}

69#[cfg(test)]
70mod tests {
71    use super::*;
72    use crate::common::types::edges_from_tuples;
73    use crate::common::utils::NodeQueue;
74
75    use crate::common::graph::Graph;
76    use crate::common::node_partitioning::FastNodePartitioning;
77
78    #[test]
79    fn test_assign_fastnodepartitioning_partition_update() {
80        // Create a simple graph: 0-1, 0-2, 1-2
81        let edges = edges_from_tuples(vec![(0, 1), (0, 2), (1, 2)]);
82        let graph = Graph::from_edges(edges.into_iter(), 3);
83
84        // Initial partition: node 0 in 0, node 1 in 1, node 2 in 2
85        let mut partition = FastNodePartitioning::initialize_with_singlet_partitions(3);
86        let mut queue = NodeQueue::from(vec![0, 1, 2]);
87        let node_i = 0;
88
89        // Assign should move node 0 to the most common neighbor partition (which is a tie, but will pick one)
90        let strategy = DefaultAssignmentStrategy;
91        strategy.assign(&graph, &mut partition, &mut queue, node_i);
92
93        // After assignment, node 0 should be in the same partition as either node 1 or node 2
94        let p0 = partition.get_partition_for_node(&0);
95        let p1 = partition.get_partition_for_node(&1);
96        let p2 = partition.get_partition_for_node(&2);
97        assert!(p0 == p1 || p0 == p2);
98
99        assert!(!queue.is_empty());
100    }
